    BEGIN;
CREATE TABLE "rest_user" (
    "user_ptr_id" integer NOT NULL PRIMARY KEY REFERENCES "auth_user" ("id"),
    "sex" varchar(1) NOT NULL
)
;
CREATE TABLE "rest_place" (
    "id" integer NOT NULL PRIMARY KEY,
    "name" varchar(200) NOT NULL,
    "lat" real NOT NULL,
    "lng" real NOT NULL,
    "address" text NOT NULL
)
;
CREATE TABLE "rest_checkin" (
    "id" integer NOT NULL PRIMARY KEY,
    "user_id" integer NOT NULL REFERENCES "rest_user" ("user_ptr_id"),
    "place_id" integer NOT NULL REFERENCES "rest_place" ("id"),
    "time" datetime NOT NULL
)
;
COMMIT;
